import { Tabs, TabItem } from '@astrojs/starlight/components';

Add `vue-simple-dnd` using your favorite package manager:

<Tabs>
    <TabItem label="npm">
    ```bash
    npm install vue-simple-dnd
    ```
    </TabItem>
    <TabItem label="yarn">
    ```bash
    yarn add vue-simple-dnd
    ```
    </TabItem>
    <TabItem label="pnpm">
    ```bash
    pnpm add vue-simple-dnd
    ```
    </TabItem>
</Tabs>
